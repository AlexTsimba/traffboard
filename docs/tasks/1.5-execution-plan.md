## Business Logic Examples (MANDATORY)

### Attribution Business Rules:
**Registration Attribution**: Include ALL lifetime value from players grouped by registration date
**FTD Attribution**: Include only FTD events grouped by the date FTD actually occurred

### Concrete Example:
```
Data:
Player A: Registration Jan 1 → FTD Jan 1 ($100)
Player B: Registration Jan 1 → FTD Jan 3 ($200) 
Player C: Registration Jan 3 → FTD Jan 3 ($150)

Registration Attribution Results:
Jan 1: count=2, ftdSum=$300 (Player A + B lifetime value)
Jan 3: count=1, ftdSum=$150 (Player C lifetime value)

FTD Attribution Results:
Jan 1: count=1, ftdSum=$100 (only Player A FTD that day)
Jan 3: count=2, ftdSum=$350 (Player B + C FTD that day)
```

### Business Rule: 
Registration Attribution answers "What's the total value of players who came on this date?"
FTD Attribution answers "How much money actually came in on this date?"

### Correct Attribution Query Logic
```typescript
// Two different query strategies based on attribution

// REGISTRATION ATTRIBUTION: Get all players by signUpDate + their lifetime FTD
async function getPartnersRegistrationAttribution(query: FilterOptionsRequest): Promise<FilterOption[]> {
  const partnersData = await prisma.playerData.groupBy({
    by: ["partnerId", "companyName"],
    where: {
      signUpDate: { not: null }, // Filter by registration date
      ...(query.search && {
        OR: [
          { partnerId: { contains: query.search, mode: "insensitive" } },
          { companyName: { contains: query.search, mode: "insensitive" } },
        ],
      }),
    },
    _sum: { 
      ftdCount: true,  // Count ALL FTD from these players
      ftdSum: true     // Sum ALL FTD from these players (regardless of FTD date)
    },
    orderBy: { _sum: { ftdCount: "desc" } },
    take: query.limit,
  });

  return partnersData.map((partner) => ({
    id: partner.partnerId,
    label: `${partner.partnerId} - ${partner.companyName ?? partner.partnerId}`,
    count: partner._sum.ftdCount ?? 0,
    ftdSum: Number(partner._sum.ftdSum ?? 0),
  }));
}

// FTD ATTRIBUTION: Get only players who made FTD, grouped by FTD date
async function getPartnersFtdAttribution(query: FilterOptionsRequest): Promise<FilterOption[]> {
  const partnersData = await prisma.playerData.groupBy({
    by: ["partnerId", "companyName"],
    where: {
      firstDepositDate: { not: null }, // Only players who made FTD
      ...(query.search && {
        OR: [
          { partnerId: { contains: query.search, mode: "insensitive" } },
          { companyName: { contains: query.search, mode: "insensitive" } },
        ],
      }),
    },
    _sum: { 
      ftdCount: true,  // Count FTD events
      ftdSum: true     // Sum FTD amounts
    },
    orderBy: { _sum: { ftdCount: "desc" } },
    take: query.limit,
  });

  return partnersData.map((partner) => ({
    id: partner.partnerId,
    label: `${partner.partnerId} - ${partner.companyName ?? partner.partnerId}`,
    count: partner._sum.ftdCount ?? 0,
    ftdSum: Number(partner._sum.ftdSum ?? 0),
  }));
}

// Main function that chooses correct strategy
async function getPartnersWithAttribution(query: FilterOptionsRequest): Promise<FilterOption[]> {
  switch (query.attribution) {
    case 'registration':
      return getPartnersRegistrationAttribution(query);
    case 'ftd':
      return getPartnersFtdAttribution(query);
    default:
      const _exhaustive: never = query.attribution;
      throw new Error(`Invalid attribution type: ${_exhaustive}`);
  }
}
```

### Edge Case Handling
```typescript
// Safe attribution query with fallback
export async function safeAttributionQuery<T>(
  attribution: 'ftd' | 'registration',
  queryFn: (dateField: string) => Promise<T[]>,
  fallbackFn?: () => Promise<T[]>
): Promise<T[]> {
  try {
    const dateField = getAttributionDateField(attribution);
    const results = await queryFn(dateField);
    
    // If no results and fallback is available, try alternative attribution
    if (results.length === 0 && fallbackFn) {
      return await fallbackFn();
    }
    
    return results;
  } catch (error) {
    console.error(`Attribution query failed for ${attribution}:`, error);
    
    if (fallbackFn) {
      return await fallbackFn();
    }
    
    throw error;
  }
}
```
